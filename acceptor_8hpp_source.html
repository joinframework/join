500 virtual Socket accept () const
501 {
502 Endpoint endpoint;
503 socklen_t addrLen = endpoint.length ();
504 Socket client (this->_tlsContext, Socket::ServerMode);
505
506 // accept connection.
507 client._handle = ::accept (this->_handle, endpoint.addr (), &addrLen);
508 if (client._handle == -1)
509 {
510 lastError = std::make_error_code (static_cast <std::errc> (errno));
511 return {};
512 }
513
514 client._state = Socket::Connected;
515
516 // set client socket mode.
517 if (client.setMode (Socket::NonBlocking) == -1)
518 {
519 client.close ();
520 return {};
521 }
522
523 // set the no delay option.
524 if (client.setOption (Socket::NoDelay, 1) == -1)
525 {
526 client.close ();
527 return {};
528 }
529
530 // create an SSL object for the connection.
531 client._tlsHandle.reset (SSL_new (client._tlsContext.get ()));
532 if (client._tlsHandle == nullptr)
533 {
534 lastError = make_error_code (Errc::OutOfMemory);
535 client.close ();
536 return {};
537 }
538
539 // set the file descriptor as the input/output facility for the TLS/SSL handle.
540 if (SSL_set_fd (client._tlsHandle.get (), client._handle) == 0)
541 {
542 lastError = make_error_code (Errc::InvalidParam);
543 client.close ();
544 return {};
545 }
546
547 // prepare the object to work in server mode.
548 SSL_set_accept_state (client._tlsHandle.get ());
549
550 // save the internal context.
551 SSL_set_app_data (client._tlsHandle.get (), &client);
552
553 #ifdef DEBUG
554 // set info callback.
555 SSL_set_info_callback (client._tlsHandle.get (), Socket::infoWrapper);
