569 static uint16_t checksum (const uint16_t* data, size_t len, uint16_t current = 0)
570 {
571 uint32_t sum = current;
572
573 while (len > 1)
574 {
575 sum += *data++;
576 len -= 2;
577 }
578
579 if (len == 1)
580 {
581 #if __BYTE_ORDER == __LITTLE_ENDIAN
582 sum += *reinterpret_cast <const uint8_t *> (data);
583 #else
584 sum += *reinterpret_cast <const uint8_t *> (data) << 8;

586 }

589 sum += (sum >> 16);

591 return static_cast <uint16_t> (~sum);
