461 BasicInternetEndpoint (const char* url) noexcept
462 : BasicEndpoint <Protocol> ()
463 {
464 // regular expression inspired by rfc3986 (see https://www.ietf.org/rfc/rfc3986.txt)
465 // ex.
466 // 0: https://example.com:8080/foo/bar.html?val=1#frag # URL
467 // 1: https # Scheme
468 // 2: example.com # Host
469 // 3: 8080 # Port
470 // 4: /foo/bar.html # Path
471 // 5: val=1 # Query
472 // 6: frag # Fragment
473 // 0 1 2 3 4 5 6
474 std::regex reg (R"(^(?:([^:/?#]+)://)?([a-z0-9\-._~%]+|\[[a-f0-9:.]+\])(?::([0-9]+))?([^?#]*)(?:\?([^#]*))?(?:#(.*))?)");
475 std::cmatch match;
476
477 this->_hostname = url;
478 uint16_t port = 0;
479
480 if (std::regex_match (url, match, reg))
481 {
482 this->_hostname = match[2];
483 this->_hostname.erase (0, this->_hostname.find_first_not_of ("["));
484 this->_hostname.erase (this->_hostname.find_last_not_of ("]") + 1);
485 port = (match[3].length ()) ? uint16_t (std::stoi (match[3])) : Resolver::resolveService (match[1]);
486 }
487
488 try
489 {
490 this->ip (this->_hostname);
491 this->port (port);
492 }
493 catch (const std::exception& ex)
494 {
495 this->ip (Resolver::resolveHost (this->_hostname));
496 this->port (port);
497 }
