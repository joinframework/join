298 int receive (HttpResponse& response)
299 {
300 // restore concrete stream.
301 this->clearEncoding ();
302
303 // read response headers.
304 if (response.readHeaders (*this) == -1)
305 {
306 return -1;
307 }
308
309 // get connection.
310 std::string connection = response.header ("Connection");
311 std::string alive = response.header ("Keep-Alive");
312
313 // check connection.
314 if (join::compareNoCase (connection, "keep-alive"))
315 {
316 size_t pos = alive.find ("timeout=");
317 if (pos != std::string::npos)
318 {
319 this->_keepTimeout = std::chrono::seconds (::atoi (alive.substr (pos + 8, alive.find (",", pos + 8)).c_str ()));
320 }
321
322 pos = alive.find ("max=");
323 if (pos != std::string::npos)
324 {
325 this->_keepMax = ::atoi (alive.substr (pos + 4, alive.find (",", pos + 4)).c_str ());
326 }
327 }
328 else if (join::compareNoCase (connection, "close"))
329 {
330 this->_keepTimeout = std::chrono::seconds::zero ();
331 this->_keepMax = 0;
332 }

342 }
343
