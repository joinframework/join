488 inline void digitsGen (DiyFp W, DiyFp Mp, uint64_t delta, char* buffer, int& length, int& k)
489 {
490 static const uint32_t kPow10[] = {1, 10, 100, 1000, 10000, 100000, 1000000, 10000000, 100000000, 1000000000};
491 DiyFp one (static_cast <uint64_t> (1) << -Mp._exponent, Mp._exponent);
492 DiyFp wp_w = Mp - W;
493 uint32_t p1 = static_cast <uint32_t> (Mp._mantissa >> -one._exponent);
494 uint64_t p2 = Mp._mantissa & (one._mantissa - 1);
495 int kappa = static_cast <int> (digitsCount (p1));
496 length = 0;
497
498 while (kappa > 0)
499 {
500 uint32_t d = 0;
501 switch (kappa)
502 {
503 case 10: d = p1 / 1000000000; p1 %= 1000000000; break;
504 case 9: d = p1 / 100000000; p1 %= 100000000; break;
505 case 8: d = p1 / 10000000; p1 %= 10000000; break;
506 case 7: d = p1 / 1000000; p1 %= 1000000; break;
507 case 6: d = p1 / 100000; p1 %= 100000; break;
508 case 5: d = p1 / 10000; p1 %= 10000; break;
509 case 4: d = p1 / 1000; p1 %= 1000; break;
510 case 3: d = p1 / 100; p1 %= 100; break;
511 case 2: d = p1 / 10; p1 %= 10; break;
512 case 1: d = p1; p1 = 0; break;
513 default: break;
514 }
515 if (d || length)
516 {
517 buffer[length++] = '0' + d;
518 }
519 --kappa;
520 uint64_t tmp = (static_cast <uint64_t> (p1) << -one._exponent) + p2;
521 if (tmp <= delta)
522 {
523 k += kappa;
524 grisuRound (buffer, length, delta, tmp, static_cast <uint64_t> (kPow10[kappa]) << -one._exponent, wp_w._mantissa);
525 return;
526 }
527 }
528
529 uint64_t unit = 1;
530 for (;;)
531 {
532 p2 *= 10; delta *= 10; unit *= 10;
533 char d = static_cast <char> (p2 >> -one._exponent);
534 if (d || length)
535 {
536 buffer[length++] = '0' + d;
537 }
538 p2 &= one._mantissa - 1;
539 --kappa;
540 if (p2 < delta)
541 {
542 k += kappa;
543 grisuRound (buffer, length, delta, p2, one._mantissa, wp_w._mantissa * unit);
544 return;
545 }
546 }
